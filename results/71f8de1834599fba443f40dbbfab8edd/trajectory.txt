function parseArxivAtom(xml){
  const entries = [...xml.matchAll(/<entry>([\s\S]*?)<\/entry>/g)].map(m=>m[1]);
  const decodeXml = s => (s||'').replace(/<!\[CDATA\[([\s\S]*?)\]\]>/g,'$1')
    .replace(/&amp;/g,'&').replace(/&lt;/g,'<').replace(/&gt;/g,'>').replace(/&quot;/g,'"').replace(/&#39;/g,"'");
  const tag = (block, name) => decodeXml((block.match(new RegExp(`<${name}[^>]*>([\\s\\S]*?)<\\/${name}>`))||[])[1]||'').replace(/\s+/g,' ').trim();
  return entries.map(b => {
    const idUrl = tag(b,'id').replace('http://','https://');
    const cats = [...b.matchAll(/<category term="([^"]+)"/g)].map(x=>x[1]);
    const authors = [...b.matchAll(/<author>\s*<name>([\s\S]*?)<\/name>\s*<\/author>/g)].map(x=>decodeXml(x[1]).trim());
    return {
      id: (idUrl.match(/abs\/(.*)$/)||[])[1] || idUrl,
      url: idUrl.replace(/v\d+$/,''),
      title: tag(b,'title'),
      updated: tag(b,'updated').slice(0,10),
      published: tag(b,'published').slice(0,10),
      categories: cats.join(', '),
      primaryCategory: (b.match(/<arxiv:primary_category[^>]*term="([^"]+)"/)||[])[1] || cats[0] || '',
      authors: authors.join(', '),
      summary: tag(b,'summary').slice(0,280)
    };
  });
}
